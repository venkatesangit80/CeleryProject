import pandas as pd
import numpy as np

# Sample DataFrame
data = {
    'A': [np.nan, 2, np.nan, 4],
    'B': [1, np.nan, 3, np.nan],
    'C': [np.nan, np.nan, np.nan, 5]
}
df = pd.DataFrame(data)

# Select the first non-null value from A, B, or C
df['ColumnA'] = df[['A', 'B', 'C']].bfill(axis=1).iloc[:, 0]

print(df)
