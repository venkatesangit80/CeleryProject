import pandas as pd

# Example DataFrame
data = {'ac_count': [1, 2, 3, 2, 1],
        'ac_value': ['passive', 'passive', 'active', 'passive', 'active']}

df = pd.DataFrame(data)

# Apply the condition
df['ac_value'] = df.apply(lambda row: 'active' if row['ac_count'] == 2 else row['ac_value'], axis=1)

print(df)
